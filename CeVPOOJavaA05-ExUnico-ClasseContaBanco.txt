package aula05poo;

public class ContaBanco {

    public int numConta;
    protected String tipo;
    private String dono;
    private float saldo;
    private boolean status;

    public int getNumConta() {
        return numConta;
    }

    public void setNumConta(int numConta) {
        this.numConta = numConta;
    }

    public String getTipo() {
        return tipo;
    }

    public void setTipo(String tipo) {
        this.tipo = tipo;
    }

    public String getDono() {
        return dono;
    }

    public void setDono(String dono) {
        this.dono = dono;
    }

    public float getSaldo() {
        return saldo;
    }

    public void setSaldo(float saldo) {
        this.saldo = saldo;
    }

    public boolean isStatus() {
        return status;
    }

    public void setStatus(boolean status) {
        this.status = status;
    }

    public ContaBanco() {
        this.saldo = 0;
        this.status = false;

    }

    public boolean abrirConta(String t) {

        setTipo(t);
        setStatus(true);

        if (t == "CC" || t == "cc" || t == "cC" || t == "Cc") {
            System.out.println("Prezado(a) " + this.dono + " \n Seu tipo de conta é: "
                    + "\n CC - Conta Corrente | CC: começa com R$ 50.00 ");
            this.saldo = 50;
        } else {
            this.tipo = "CP";
            System.out.println("Prezado(a) " + this.dono
                    + " \n CP - Conta Poupança | CP: Começa com R$ 150.00");
            this.saldo = 150;

        }
        return this.status;
    }

    public boolean fecharConta() {
        if (this.saldo < 0) {
            System.out.println("ERRO ao fechar a conta: "
                    + "Você possui em débito R$" + this.saldo);
            this.status = true;
        } else if (this.saldo > 0) {
            System.out.println("ERRO ao fechar a conta: "
                    + "Você possui R$" + this.saldo);
            this.status = true;
        } else {
            System.out.println("SUCESSO ao fechar a conta.");
            this.status = false;
        }

        return this.status;
    }

    public float depositar(float dep) {
        if (this.status == true) {
            this.saldo = this.saldo + dep;
            // op. alt.: setSaldo( getSaldo() + dep );
        } else {
            System.out.println("É impossível sacar.");
        }
        return this.saldo;
    }

    public float sacar(float saque) {
        if (this.status == true) {
            if (saque <= this.saldo) {
                this.saldo = this.saldo - saque;
            } else {
                System.out.println("ERRO ao sacar: "
                        + "Valor acima do saldo disponível.");
            }
        } else {
            System.out.println("Impossível sacar.");
        }

        return this.saldo;
    }

    public void pagarMens() {
        float val_mens;

        if (this.tipo == "CC") {
            // this.saldo = this.saldo - 12;
            val_mens = 12;
        } else {
            // this.saldo = this.saldo - 20;
            val_mens = 20;
        }
        if (this.status == true) {
            if (this.saldo > val_mens) {
                this.saldo = this.saldo - val_mens;
            } else {
                System.out.println("Saldo insuficiente.");
            }
        } else {
            System.out.println("Impossível pagar.");
        }

    }

}
